def preprocess(file_path):
    UWB_cols=['BTM_UWBDst_Dev1', 'BTARR_UWBDst_Dev1','BTAL_UWBDst_Dev1', 'BTAR_UWBDst_Dev1','BTARB_UWBDst_Dev1']
    RSSI_cols=['BTM_RSSI_Dev1', 'BTARR_RSSI_Dev1','BTAL_RSSI_Dev1', 'BTAR_RSSI_Dev1', 'BTARB_RSSI_Dev1']
    
    data=pd.read_csv(file_path)
    
    #处理BTARB单位
    data['BTARB_RSSI_Dev1']=data['BTARB_RSSI_Dev1'].apply(lambda x:x-255)
    
    #处理RSSI -255和UWB 0以及UWB>1200的极值
    for RSSI_col in RSSI_cols:data[RSSI_col]=data[RSSI_col].replace(-255.0,np.nan)
    for UWB_col in UWB_cols:data[UWB_col]=data[UWB_col].apply(lambda x:x if 0<x<1200 else np.nan)
        
    #上一就近填补Nan
    data.fillna(method='pad',limit=40,inplace=True)
    
    #空缺过大，予以丢弃
    data.dropna(how='any',inplace=True)

    #打标签
    #data['Location']='后备箱'
    
    return data

def compute_acc(prediction,data_test,window_width):
    print('正确率:',100-100*sum(prediction.values[window_width-1:]!=data_test['Location'].values[window_width-1:])/len(prediction[window_width-1:]),'%')
    
def test(data_test,filter_window=40,title=None):
    
    y_pred=model.predict(data_test[['BTM_RSSI_Dev1', 'BTM_UWBDst_Dev1', 'BTARR_RSSI_Dev1', 'BTARR_UWBDst_Dev1', 'BTAL_RSSI_Dev1', 'BTAL_UWBDst_Dev1', 'BTAR_RSSI_Dev1', 'BTARB_RSSI_Dev1', 'BTAR_UWBDst_Dev1', 'BTARB_UWBDst_Dev1']])
    y_pred=pd.Series(y_pred,index=list(range(len(y_pred))))
    

    window_width=filter_window
    y_pred_filtered=y_pred.rolling(window=window_width).apply(lambda x: mode(x)[0])

    plt.figure(figsize=(10,6))
    plt.plot(list(range(len(y_pred))),y_pred,'b',color='grey',label='Pred',alpha=0.2,linewidth=0.5)
    plt.plot(list(range(len(y_pred))),y_pred_filtered,'bo',color='coral',label='Filtered',marker='.',alpha=0.5)
    plt.plot(list(range(len(y_pred))),data_test['Location'],'b',color='steelblue',label='True')
    if title: plt.title(title)
    plt.legend()
    plt.show()
    
    print('单点')
    compute_acc(y_pred,data_test,window_width=filter_window)
    
    print('滤波后')
    compute_acc(y_pred_filtered,data_test,window_width=filter_window)
